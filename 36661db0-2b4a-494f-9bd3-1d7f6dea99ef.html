    <script>
        // Función para que el botón "No" escape del cursor
        function huir() {
            const btnNo = document.getElementById('btn-no');
            // Genera posiciones aleatorias dentro de la pantalla, cuidando que no se salga
            const x = Math.random() * (window.innerWidth - btnNo.offsetWidth - 40);
            const y = Math.random() * (window.innerHeight - btnNo.offsetHeight - 40);
            
            // Cambia la posición del botón de manera absoluta en toda la pantalla
            btnNo.style.position = 'fixed';
            btnNo.style.left = x + 'px';
            btnNo.style.top = y + 'px';
        }

        // Pasar a la siguiente pantalla al dar "Sí"
        function aceptar() {
            document.getElementById('pantalla1').classList.add('hidden');
            document.getElementById('pantalla2').classList.remove('hidden');
        }

        // Acción al finalizar
        function enviarDetalles() {
            const fecha = document.getElementById('fecha').value;
            const hora = document.getElementById('hora').value;
            const planSelect = document.getElementById('plan');
            const planTexto = planSelect.options[planSelect.selectedIndex].text;

            if(!fecha || !hora) {
                alert("Por favor, selecciona una fecha y una hora para que sea oficial. 😏");
                return;
            }

            // Muestra un mensaje final confirmando los datos que eligió
            alert(`¡Trato hecho! Nos vemos el ${fecha} a las ${hora}.\nElegiste: ${planTexto}.\n\n¡No olvides tu recordatorio de 3 días antes!`);
            
            // Opcional: Aquí podrías redirigirla a tu WhatsApp con los datos ya escritos si quisieras.
        }
